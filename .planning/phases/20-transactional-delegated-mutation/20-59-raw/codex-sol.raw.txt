   141	/// fan-out, scoped to the test's own tagged process tree rather than host-wide
   142	/// by image name.

   144	/// A bare `choice.exe` idler cannot carry an injected tag on its own command

   147	/// `start "" /b choice` idler is a direct child of that cmd, so its
   148	/// `ParentProcessId` is the tagged cmd's `ProcessId`. Selecting only choice

   150	/// `choice`-spawning target on the same runner (e.g. `live_fs_acl`) cannot
   151	/// pollute the capture.

   154	/// are captured WHILE the tagged parent cmd is still alive (during the
   155	/// peak-sampling window), and the `ParentProcessId` scope is what makes that
   156	/// capture immune to a concurrent `choice`-spawning target. Once the job closes
   157	/// the tagged parent is dead, so this parent-scoped query would go structurally
   158	/// empty regardless of a leaked survivor — the post-close survivor check is
   159	/// therefore done by fixed ProcessId via [`surviving_captured_choice_pids`],
   160	/// NOT by re-running this parent-scoped query.

 /// Poll `predicate` up to `deadline_secs`, panicking with `message` on timeout.
 /// Mirrors the `wait_until` helper in `live_fs_acl.rs`.
 fn wait_until(mut predicate: impl FnMut() -> bool, deadline_secs: u64, message: &str) {
     let deadline = Instant::now() + Duration::from_secs(deadline_secs);
     while Instant::now() < deadline {
         if predicate() {
             return;
         }
         std::thread::sleep(Duration::from_millis(100));
     }
     panic!("timed out waiting for {message}");
 }

   137	/// True when a resolved `lpApplicationName` (NUL-terminated UTF-16, as
   138	/// returned by [`resolve_program`]) has `cmd.exe` as its exact FINAL PATH
   139	/// COMPONENT (case-insensitive) — NOT a bare suffix, so a name like
   140	/// `notcmd.exe`/`foocmd.exe` does not match. Used to decide whether the
   141	/// `/c`/`/k` payload needs the cmd-aware quoting above rather than the general
   142	/// CRT [`quote_arg`] rules. Matching the RESOLVED program (not the raw
   143	/// `argv[0]`) is deliberate: bare `cmd`/`cmd.exe` is pinned to
   144	/// `System32\cmd.exe` and an absolute path is validated as-is, so this reflects
   145	/// the actual image that will parse the command line.
   146	pub(super) fn resolved_program_is_cmd(app_name_w: &[u16]) -> bool {
   147	    let end = app_name_w
   148	        .iter()
   149	        .position(|&c| c == 0)
   150	        .unwrap_or(app_name_w.len());
   151	    // Windows `Path` treats both `\` and `/` as separators, so `file_name`
   152	    // isolates the true final component of the `System32\cmd.exe` resolution or
   153	    // any absolute path the caller passed; equality (not a suffix) rejects
   154	    // `notcmd.exe`.
   155	    let decoded = String::from_utf16_lossy(&app_name_w[..end]).to_ascii_lowercase();
   156	    std::path::Path::new(&decoded)
   157	        .file_name()
   158	        .and_then(|name| name.to_str())
   159	        .is_some_and(|name| name == "cmd.exe")
   160	}

   217	                SwarmError::DispatchAdmission("retained worktree count overflowed".into())
   218	            })?;
